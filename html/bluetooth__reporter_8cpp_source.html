176 bool BluetoothReporter::send_reading(const Reading& reading) const {
177  if(_pServer->getConnectedCount() == 0) {
178  // No clients to send data to
179  return false;
180  }
181  DEBUG_PRINTLN("Sending reading over Bluetooth");
182  const char* reading_str = reading.get_reading_str();
183  size_t size = strlen(reading_str);
184 
185  int segment = 0;
186  const static uint8_t max_segment_size = 20; // Max that can be send over bluetooth
187  do {
188  ++segment;
189  uint8_t segment_size = segment * max_segment_size > size ? size % max_segment_size : max_segment_size;
190  char to_send[segment_size];
191  strncpy(to_send, reading_str + ((segment - 1) * max_segment_size), segment_size);
192 
193  pDataRXCharacteristic->setValue((uint8_t*) to_send, segment_size);
194  pDataRXCharacteristic->notify();
195  } while(segment * max_segment_size < size);
196  return true;
197 }

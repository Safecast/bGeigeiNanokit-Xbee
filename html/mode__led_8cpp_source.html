51 void ModeLED::loop() {
52  if(frequency > 0) {
53  // Blink LED
54  double blink_millis = 1000 / frequency;
55  auto cycle_now = millis() % static_cast<uint32_t>(blink_millis);
56  auto threshold = static_cast<uint32_t>(blink_millis * (percentage_on / 100.0));
57  if(cycle_now < threshold) {
58  set_color(color);
59  } else if(cycle_now > threshold) {
60  set_color(ModeLED::mode_color_off);
61  }
62  }
63 }

74 void ModeLED::handle_report(const Report& report) {
75  auto& worker_stats = report.get_worker_stats();
76  auto& handler_stats = report.get_handler_stats();
77 
78  // Switch controller system state
79  switch(handler_stats.at(k_handler_controller_handler).status) {
80  case ControllerState::k_state_InitializeState:
81  case ControllerState::k_state_InitReadingState:
83  set_values(mode_color_init, 1, 10);
84  break;
85  case ControllerState::k_state_PostInitializeState:
87  set_values(mode_color_init, 1.5, 15);
88  break;
89  case ControllerState::k_state_ConfigurationModeState:
90  switch(worker_stats.at(k_worker_configuration_server).active_state) {
91  case WorkerStatus::e_state_active:
93  set_values(mode_color_config);
94  break;
95  case WorkerStatus::e_state_activating_failed:
96  default:
98  set_values(mode_color_config, 1, 50);
99  break;
100  }
101  break;
102  case ControllerState::k_state_MobileModeState:
103  switch(handler_stats.at(k_handler_bluetooth_reporter).status) {
104  case BluetoothReporter::e_handler_clients_available:
106  set_values(mode_color_mobile);
107  break;
108  case BluetoothReporter::e_handler_idle:
109  case BluetoothReporter::e_handler_no_clients:
110  default:
112  set_values(mode_color_mobile, 0.33, 5);
113  break;
114  }
115  break;
116  case ControllerState::k_state_FixedModeState:
117  if(handler_stats.at(k_handler_api_reporter).active_state == HandlerStatus::e_state_activating_failed) {
119  set_values(mode_color_fixed_connected, 1, 25);
120  break;
121  }
122  switch(handler_stats.at(k_handler_api_reporter).status) {
123  case ApiReporter::e_api_reporter_error_not_connected:
125  set_values(mode_color_fixed_connected, 1, 25);
126  break;
127  case ApiReporter::e_api_reporter_error_invalid_reading:
129  set_values(mode_color_fixed_hard_error, 0.33, 15);
130  break;
131  case ApiReporter::e_api_reporter_error_remote_not_available:
132  case ApiReporter::e_api_reporter_error_server_rejected_post:
134  set_values(mode_color_fixed_soft_error);
135  break;
136  case ApiReporter::e_api_reporter_send_success:
137  default:
139  set_values(mode_color_fixed_connected);
140  break;
141  }
142  break;
143  case ControllerState::k_state_ResetState:
145  set_values(mode_color_fixed_hard_error);
146  break;
147  default:
148  break;
149  }

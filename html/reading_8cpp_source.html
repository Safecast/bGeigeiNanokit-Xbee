32 double calc_distance(double lon1, double lat1, double lon2, double lat2) {
33  //This portion converts the current and destination GPS coords from decDegrees to Radians
34  lon1 *= (PI / 180);
35  lon2 *= (PI / 180);
36  lat1 *= (PI / 180);
37  lat2 *= (PI / 180);
38 
39  //This portion calculates the differences for the Radian latitudes and longitudes and saves them to variables
40  double dlon = lon2 - lon1;
41  double dlat = lat2 - lat1;
42 
43  //This portion is the Haversine Formula for distance between two points. Returned value is in KM
44  double a = (sq(sin(dlat / 2))) + cos(lat1) * cos(lat2) * (sq(sin(dlon / 2)));
45  double e = 2 * atan2(sqrt(a), sqrt(1 - a));
46  return 6371.00 * e;
47 }

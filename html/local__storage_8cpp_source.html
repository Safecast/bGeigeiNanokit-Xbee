290bool LocalStorage::activate(bool) {
291 _memory.begin(memory_name, true);
292 _device_id = _memory.getUShort(key_device_id, D_DEVICE_ID);
293 if(_memory.getString(key_ap_password, _ap_password, CONFIG_VAL_MAX) == 0) {
294 strcpy(_ap_password, D_ACCESS_POINT_PASSWORD);
295 }
296 if(_memory.getString(key_wifi_ssid, _wifi_ssid, CONFIG_VAL_MAX) == 0) {
297 strcpy(_wifi_ssid, D_WIFI_SSID);
298 }
299 if(_memory.getString(key_wifi_password, _wifi_password, CONFIG_VAL_MAX) == 0) {
300 strcpy(_wifi_password, D_WIFI_PASSWORD);
301 }
302 if(_memory.getString(key_api_key, _api_key, CONFIG_VAL_MAX) == 0) {
303 strcpy(_api_key, D_APIKEY);
304 }
305 _use_dev = _memory.getBool(key_use_dev, D_USE_DEV_SERVER);
306 _led_color_blind = _memory.getBool(key_led_color_blind, D_LED_COLOR_BLIND);
307 _led_color_intensity = _memory.getUChar(key_led_color_intensity, D_LED_COLOR_INTENSITY);
308 _saved_state = _memory.getChar(key_saved_state, D_SAVED_STATE);
309 _use_home_location = _memory.getBool(key_led_color_blind, false);
310 _home_longitude = _memory.getDouble(key_home_longtitude, 0);
311 _home_latitude = _memory.getDouble(key_home_latitude, 0);
312 _last_longitude = _memory.getDouble(key_last_longtitude, 0);
313 _last_latitude = _memory.getDouble(key_last_latitude, 0);
314 _memory.end();
315 return true;
316}

318int8_t LocalStorage::handle_produced_work(const worker_status_t& worker_reports) {
319 // Get reading data to store
320 const auto& reader = worker_reports.at(k_worker_bgeigie_connector);
321 if(reader.is_fresh()) {
322 const auto& reading = reader.get<Reading>();
323 set_device_id(reading.get_device_id(), false);
324 if(reading.get_status() & k_reading_gps_ok) {
325 set_last_latitude(reading.get_latitude(), false);
326 set_last_longitude(reading.get_longitude(), false);
327 }
328 return HandlerStatus::e_handler_data_handled;
329 }
330 return HandlerStatus::e_handler_idle;
